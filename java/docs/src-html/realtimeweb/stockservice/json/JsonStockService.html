001    package realtimeweb.stockservice.json;
002    
003    import java.util.HashMap;
004    
005    import realtimeweb.stockservice.main.AbstractStockService;
006    import realtimeweb.stockservice.util.Util;
007    
008    /**
009     * Used to get data as a raw string.
010     */
011    public class JsonStockService implements AbstractStockService {
012            private static JsonStockService instance;
013            protected boolean local;
014            private ClientStore clientStore;
015            /**
016             * **For internal use only!** Protected Constructor guards against instantiation.
017            
018             * @return 
019             */
020            protected  JsonStockService() {
021                    disconnect();
022                    this.clientStore = new ClientStore();
023            }
024            
025            /**
026             * Retrieves the singleton instance.
027            
028             * @return JsonStockService
029             */
030            public static JsonStockService getInstance() {
031                    if (instance == null) {
032                            synchronized (JsonStockService.class) {
033                                    if (instance == null) {
034                                            instance = new JsonStockService();
035                                    }
036                            }
037                            
038                    }
039                    return instance;
040            }
041            
042            /**
043             * Establishes a connection to the online service. Requires an internet connection.
044            
045             */
046            @Override
047            public void connect() {
048                    this.local = false;
049            }
050            
051            /**
052             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
053            
054             */
055            @Override
056            public void disconnect() {
057                    this.local = true;
058            }
059            
060            /**
061             * **For internal use only!** The ClientStore is the internal cache where offline data is stored.
062            
063             * @return ClientStore
064             */
065            public ClientStore getClientStore() {
066                    return this.clientStore;
067            }
068            
069            /**
070             * Retrieves current stock information.
071             * @param ticker A comma separated list of ticker symbols (e.g. "AAPL, MSFT, CSCO").
072             * @return String
073             */
074            public String getStockInformation(String ticker) throws Exception {
075                    String url = String.format("http://www.google.com/finance/info", ticker);
076                    HashMap<String, String> parameters = new HashMap<String, String>();
077                    parameters.put("q", String.valueOf(ticker));
078                    parameters.put("client", String.valueOf("iq"));
079                    if (this.local) {
080                            return clientStore.getData(Util.hashRequest(url, parameters));
081                    }
082                    String jsonResponse = "";
083                    try {
084                        jsonResponse = Util.get(url, parameters);
085                        if (jsonResponse.startsWith("<")) {
086                            throw new Exception(jsonResponse);
087                        }
088                        return jsonResponse;
089                    } catch (Exception e) {
090                        throw new Exception(e.toString());
091                    }
092            }
093            
094            /**
095             * Retrieves current stock information.
096             * @param ticker A comma separated list of ticker symbols (e.g. "AAPL, MSFT, CSCO").
097             * @param callback The listener that will be given the data (or error).
098             */
099            public void getStockInformation(final String ticker, final JsonGetStockInformationListener callback) {
100                    
101                    Thread thread = new Thread() {
102                        @Override
103                        public void run() {
104                            try {
105                                callback.getStockInformationCompleted(JsonStockService.getInstance().getStockInformation(ticker));
106                            } catch (Exception e) {
107                                callback.getStockInformationFailed(e);
108                            }
109                        }
110                    };
111                    thread.start();
112                    
113            }
114            
115    }
